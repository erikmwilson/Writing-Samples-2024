"disabled": false,
"confd\_hash": null,
"criteria": ":set consecutiveCount=3 if \
  (metric[\\"nova-compute\_status\\"] != 1) \
  { return new AlarmStatus(CRITICAL, \\"nova-compute down\\"); }",
...
